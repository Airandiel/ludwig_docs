[docs]def read_characteristics(path="char_gold_unprotected.txt"):
    """
    Reads characteristic from text file, file have to consist two columns, one with wavelength, second with intensity,
    usually from 0 to 1, file can have headers, but they have to begin with "[" mark

    :param str path: path to file
    :return: (np.array) read data
    """
    with open(path, "r") as file:
        lines = file.readlines()
        data = []
        for line in lines:
            if line[0] != '[':
                words = re.split(',|\t| |:|\n', line)
                data.append([float(words[0]), float(words[1])])

    return np.array(data)




[docs]def normalize_characteristics(data, shift_maximum=0):
    """
    Normalize data, 2D array, with two columns, it normalizes the second column, shift_maximum can shift maximum by some value

    :param np.array data: 2 column np.array
    :param float shift_maximum: Optional, how much shift maximum, default 0
    :return: (np.array) data
    """
    data_normalized = np.array(data)
    data_normalized[:, 1] -= np.min(data_normalized[:, 1])  # shifting to positive
    data_normalized[:, 1] /= np.max(
        data_normalized[:, 1]) + shift_maximum  # unification, plus 0.05 to not exceed 0.95% of reflectance
    return data_normalized

[docs]def log_interp1d(xx, yy, kind='linear'):
    """
    Logarithmic interpolation, returns lambda with interpolation
    :param np.array xx: X values
    :param np.array yy: Y values
    :param str kind: Optional, Kind of interpolation, default 'linear', can be 'cubic' the rest described in sp.interpolate.interp1d documentation
    :return: callable log_interp
    """
    logx = np.log10(xx)
    logy = np.log10(yy)
    lin_interp = sp.interpolate.interp1d(logx, logy, kind=kind)
    log_interp = lambda zz: np.power(10.0, lin_interp(np.log10(zz)))
    return log_interp

[docs]def linear_interpolation(data, part_to_extrapolate=2):
    """
    Linear interpolation with extrapolation, for extrapolation it uses only some part of characteristics,
    part_to_extrapolate determines the divider how much of lasts samples are taken into consideration,
    for example 2 means half, 5 means one fifth of last samples

    :param np.array data: 2 column np.array
    :param float part_to_extrapolate: divider which determines how much samples are taken into extrapolation
    :return: (callable) interpolation
    """
    x_existing = data[:, 0]
    y_existing = data[:, 1]
    half_of_length = len(x_existing) // part_to_extrapolate
    m = (half_of_length * np.sum(x_existing[-half_of_length:] * y_existing[-half_of_length:]) -
         np.sum(x_existing[-half_of_length:]) * np.sum(y_existing[-half_of_length:])) / \
        (half_of_length * np.sum(x_existing[-half_of_length:] ** 2) - np.sum(x_existing[-half_of_length:]) ** 2)
    b = (np.sum(y_existing[-half_of_length:]) - m * np.sum(x_existing[:half_of_length])) / half_of_length

    # lin_interp = sp.interpolate.interp1d(x, y, kind='linear', fill_value='extrapolate')
    # lin_interp = lambda xx: extrap(xx, x, y, m, b)
    def lin_extrap(x_extr):
        y_extr = np.interp(x_extr, x_existing, y_existing)
        y_extr = np.where(x_extr < x_existing[0], linear_regression(x_extr, m, b), y_extr)
        y_extr = np.where(x_extr > x_existing[-1], linear_regression(x_extr, m, b), y_extr)
        y_extr[y_extr < 0] = 0
        return y_extr

    return lin_extrap




[docs]def extrap(x, xp, yp, m, b):
    """
    np.interp function with linear extrapolation

    :param np.array x: Xs which we are looking for
    :param np.array xp: Xs which are defined
    :param np.array yp: Ys which are defined
    :param float m: coefficient of linear function, mx + b
    :param float b: coefficient of linear function, mx + b
    :return: (np.array) with extrapolated data
    """
    y = np.interp(x, xp, yp)
    y = np.where(x < xp[0], linear_regression(x, m, b), y)
    y = np.where(x > xp[-1], linear_regression(x, m, b), y)
    return y




[docs]def linear_regression(x, m, b):
    """
    Linear function, with parameters
    :param np.array x: Xs
    :param float m: coefficient of linear function, mx + b
    :param float b: coefficient of linear function, mx + b
    :return: (np.array) value of function
    """
    return x * m + b




[docs]def sinusoid(freq, amplitude, phase, x):
    """
    Creates sinusoid with specific parameters, types depends on input, are flexible, can be np.array, can be numbers

    :param freq:
    :param amplitude:
    :param phase:
    :param x:
    :return: value of sinusoid
    """
    return amplitude * np.sin(2 * np.pi * freq * x + phase)




[docs]def gaussian(x, mu, sig):
    """
    generate gaussian peak with some parameters, for some Xs values

    :param np.array x: Xs for which we want to generate values
    :param float mu:
    :param float sig:
    :return: (np.array) values of gaussian
    """
    return np.exp(-np.power(x - mu, 2.) / (2 * np.power(sig, 2.)))

[docs]def generate_gaussian_spectrum(lowest_x, highest_x, mu=1.5, sig=0.1, no_of_values=128, plot=False):
    """
    Wrapped function 'gaussian', for generating gaussian spectrum, in some specific range of wavelengths

    :param float lowest_x: lower border of Xs
    :param float highest_x: upper border of Xs
    :param float mu: Optional, properties of gaussian, default 1.5
    :param float sig: Optional properties of gaussian, default 0.1
    :param int no_of_values: Optional, number of points in spectrum, default 128
    :param bool plot: plotting flag, default False
    :return: (np.array) data, 2 column array
    """
    data = np.zeros([no_of_values, 2])
    data[:, 0] = np.linspace(lowest_x, highest_x, no_of_values)
    data[:, 1] = gaussian(data[:, 0], mu, sig)
    if plot:
        plt.plot(data[:, 0], data[:, 1])
    return data

    """
    Returns lambda of line described by two angles and starting point, xy angle, like in spherical coordinate system,
    xz angle like in spherical angle, but it is azimuth - pi/2
    :param float angle_xy: single XY angle
    :param float angle_xz: single XZ angle
    :param np.array starting_point: single starting point
    :return: callable line
    """
    return lambda x: ((x - starting_point[0]) * np.tan(angle_xy) + starting_point[1],
                      (x - starting_point[0]) * np.tan(angle_xz) + starting_point[2])
