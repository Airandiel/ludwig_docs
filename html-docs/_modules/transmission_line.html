[docs]    def update(self):
        """
        Calculate positions and angles of all elements of line

        :return: none
        """
        current_x = 0
        current_y = 0
        current_angle = 0
        previous_obj = TransmissionObject
        for index, obj in enumerate(self):
            current_angle, current_x, current_y = obj.construct_obj(
                current_angle, current_x, current_y, self.beginning[2])
            if index > 0:
                obj.set_distances_and_characteristics(previous_obj)
            previous_obj = obj



[docs]    def get_transformation_of_whole_line(self, wavelength):
        """
        Returns spectrum transformation of whole line, for certain wavelengths, not tested on unsorted data

        :param np.array wavelength: usually spectrum wavelengths, from min to max
        :return: (np.array) spectrum transformation
        """
        length = [0]
        spectrum = np.ones(len(wavelength))
        for obj in self:
            spectrum *= obj.get_transformation(length)(wavelength)

        self.total_length = length[0]

        if settings.air_rays_air_interaction:
            char = Behapi(temperature_celsius=settings.air_temperature_celsius,
                          pressure_pa=settings.air_pressure_pa,
                          relative_humidity=settings.air_relative_humidity, atmosphere='air')
            char.absorption_coefficient(method='sdvoigt', wavelength_range_um=[settings.wav_min, settings.wav_max],
                                        wavelength_step_um=(settings.wav_max - settings.wav_min) /
                                                           settings.no_of_spectrum_points)

            air_transmission = char.transmittance(distance_cm=length[0] / 10,
                                                  desired_wavelengths=wavelength)
            spectrum *= air_transmission
        return spectrum

    def __init__(self, transmission_line):
        self.transmission_line = transmission_line
        acceptance_angles_of_tran_line = transmission_line[1].calculate_acceptance_angle(return_min_max=True)
        self.spectrum = input_gen.generate_input_spectrum(acceptance_angles_of_tran_line)

[docs]    def add_ray(self, position, anglexy, anglexz, wavelength=-1):
        """
        Adds single ray to others, when wavelength == -1 then it adds ray for every wavelength, defined in settings,
        in the other case it adds single ray with certain wavelength

        :param np.array position: [x,y,z] starting position of ray
        :param float anglexy:
        :param float anglexz:
        :param float wavelength:
        :return: none
        """
        if wavelength == -1:
            wavelengths = self.spectrum[:, 0]
            if len(self.positions) == 0:
                self.positions = np.full((len(wavelengths), 3), position)
            else:
                self.positions = np.append(self.positions, np.full((len(wavelengths), 3), position), axis=0)
            self.anglesXY = np.append(self.anglesXY, np.full((len(wavelengths)), anglexy), axis=0)
            self.anglesXZ = np.append(self.anglesXZ, np.full((len(wavelengths)), anglexz), axis=0)
            self.wavelengths = np.append(self.wavelengths, wavelengths, axis=0)
            self.energies = np.append(self.spectrum[:, 1], self.energies, axis=0)
            self.trace_length_in_other = np.append(self.trace_length_in_other, np.zeros((len(wavelengths))), axis=0)
        else:
            self.positions = np.append(self.positions, [position], axis=0)
            self.anglesXY = np.append(self.anglesXY, [anglexy], axis=0)
            self.anglesXZ = np.append(self.anglesXZ, [anglexz], axis=0)
            self.wavelengths = np.append(self.wavelengths, [wavelength], axis=0)
            self.trace_length_in_other = np.append(self.trace_length_in_other, [0], axis=0)



[docs]    def add_rays(self, positions, anglexy, anglexz):
        """
        Create array of rays, erasing previous if were any, from every position which is given it is creating many rays
        of different wavelengths, they are defined by ranges and number of points between in settings

        :param np.array positions:
        :param np.array anglexy:
        :param np.array anglexz:
        :return: none
        """
        wavelengths = self.spectrum[:, 0]
        self.positions = np.repeat(positions, len(wavelengths), axis=0)
        self.anglesXY = np.repeat(anglexy, len(wavelengths), axis=0)
        self.anglesXZ = np.repeat(anglexz, len(wavelengths), axis=0)
        self.wavelengths = np.tile(wavelengths, (len(positions)))
        self.energies = np.tile(self.spectrum[:, 1], (len(positions)))
        self.trace_length_in_other = np.zeros((len(self.positions)))

        """
        Calculate paths of all rays, returns 3D array with points of collisions on every component, in settings can be
        set to erase not fitting rays, which goes outside mirror, transforms energies of rays

        :return: (np.array) trace_points
        """
        self.trace_points = np.array([self.positions])
        i = 0
        for obj in self.transmission_line:
            fitting_mask = obj.interact_with_obj_vectorized(self)
            # obj.transform_energies_vectorized(self)     # transform energies
            if settings.show_all_rays:
                fitting_mask = np.ones((len(self.positions)), dtype=bool)
            # if not break_flag:
            #     break
            print("Current mirror: ", i, "fitting: ", np.sum(fitting_mask))
            i += 1
            self.trace_points = np.append(self.trace_points, [self.positions], axis=0)
            self.anglesXY = self.anglesXY[fitting_mask]
            self.anglesXZ = self.anglesXZ[fitting_mask]
            self.positions = self.positions[fitting_mask, :]
            self.wavelengths = self.wavelengths[fitting_mask]
            self.trace_points = self.trace_points[:, fitting_mask, :]
            self.energies = self.energies[fitting_mask]

            #                              + (self.trace_points[-2, 1] - self.position[1]) ** 2)
        # if settings.air_rays_air_interaction:
        #     self.calculate_transition_characteristics()
        return self.trace_points

        """
        Method to plot transverse cross section of rays on certain element of transmission line, usually used to plot
        rays on screen, but can plot on other elements as well, as far as obj.plot_cross_section_on_obj() is defined
        as far it is not well defined for all elements.

        Note: Does not create new figure, just taking axes from current one.

        :param int num_of_obj_in_list: number of element of transmission line
        :return: none
        """
        ax1 = plt.axes()
        obj = self.transmission_line[num_of_obj_in_list]
        ax1 = obj.plot_cross_section_on_obj(ax1)
        plane_cross_counter = 1
        for i in range(num_of_obj_in_list):
            if self.transmission_line[i].type == 'Prism':
                plane_cross_counter += 1
        xs = self.trace_points[num_of_obj_in_list + plane_cross_counter, :, 0] - obj.position[0]
        ys = self.trace_points[num_of_obj_in_list + plane_cross_counter, :, 1] - obj.position[1]
        # which side of center
        d = xs * (obj.normal_vector[1]) - ys * (obj.normal_vector[0])
        signation = -np.sign(d)
        # np.cos(np.pi - obj.angle)
        coords_x = signation * \
                   tools.get_euclidean_distance(obj.position[:2],
                                                self.trace_points[num_of_obj_in_list + plane_cross_counter, :, :2])
        # print(coords_x)
        ax1.scatter(coords_x, self.trace_points[num_of_obj_in_list + 2, :, 2],
                    c=self.color_map_rgba(self.wavelengths), marker='*')

[docs]    def count_rays_in_pixels(self):
        """
        Counts how many each rays fall on which pixels on screen, returns 2d histogram, first axis - num of pixel,
        second axis - wavelength, in second returned variable are bins for wavelengths

        :return: (np.array, np.array) 2d histogram, wavelength bins of histogram
        """
        screen = self.transmission_line[-1]
        # sort rays by distance from left side
        index_offset = 1
        if settings.screen_show_behind:
            index_offset += 1

        xs = self.trace_points[-index_offset, :, 0] - screen.position[0]
        ys = self.trace_points[-index_offset, :, 1] - screen.position[1]
        # which side of center
        d = xs * (screen.normal_vector[1]) - ys * (screen.normal_vector[0])
        signation = -np.sign(d)
        coords_x = signation * tools.get_euclidean_distance(
            screen.position[:2], self.trace_points[-index_offset, :, :2])
        sorted_indices = np.argsort(coords_x)
        coords_x = coords_x[sorted_indices]
        energies = self.energies[sorted_indices]
        wavelengths = self.wavelengths[sorted_indices]
        cells_ranges = np.linspace(-screen.diameter / 2, screen.diameter / 2, settings.no_of_pixels + 1)
        ranges_in_coords = np.searchsorted(coords_x, cells_ranges)
        spectral_histogram = np.zeros((settings.no_of_pixels, settings.no_of_spectrum_points))
        bins = np.zeros((settings.no_of_spectrum_points + 1))
        for i in range(settings.no_of_pixels):
            current_wavs = wavelengths[ranges_in_coords[i]:ranges_in_coords[i + 1]]
            current_energies = energies[ranges_in_coords[i]:ranges_in_coords[i + 1]]
            hist, bins = np.histogram(current_wavs, bins=np.linspace(
                settings.wav_min, settings.wav_max, settings.no_of_spectrum_points + 1), weights=current_energies)
            spectral_histogram[i, :] = hist
        return spectral_histogram, bins
