    """
    Class Behapi consist of methods for using data from HITRAN database for calculating transmittance spectrum
    for object with defined temperature, pressure, relative humidity and gas mixture.

    Example of use::

        char = Behapi(temperature_celsius=20, pressure_pa=101325,
                  relative_humidity=20, atmosphere='nitrogen')
        char.absorption_coefficient(method='sdvoigt', wavelength_range_um=[0.5, 50], wavelength_step_um=0.00005)
        # deswave = np.arange(2.5, 20, 17.5/2000)
        deswave = np.arange(0.5, 50, 0.00005)
        plt.figure()
        plt.title(
            'Transmittance spectrum for nitrogen environment with 20Â°C temperature, 1013.25Pa pressure, 20% relative humidity and 30cm beam path',
            fontsize=15)
        plt.xlabel('Wavelength [$\mu m$]', fontsize=13)
        plt.ylabel('Transmittance', fontsize=13)
        plt.plot(deswave, char.transmittance(distance_cm=30, desired_wavelengths=deswave))
        plt.savefig('transmittance.svg')
    """
    def __init__(self, temperature_celsius, pressure_pa, relative_humidity, atmosphere):
        """
        :param float temperature_celsius: temperature of environment in celsius (required)
        :param float pressure_pa: pressure of environment in pascal (required)
        :param float relative_humidity: relative humidity of environment in percent (required)
        :param string atmosphere: atmosphere of environment - 'air' or 'nitrogen' (required)
        """
        self.temperature_kelvin = temperature_celsius + 273.15
        self.pressure_atm = pressure_pa / 101325
        self.relative_humidity = relative_humidity
        self.atmosphere = atmosphere

        # calculating actual amount of water in the atmosphere
        critical_pressure_pa = 22.064 * 10 ** 6
        critical_temperature_kelvin = 647.096
        temp = 1 - self.temperature_kelvin / critical_temperature_kelvin
        a1 = -7.85951783
        a2 = 1.84408259
        a3 = -11.7866497
        a4 = 22.6807411
        a5 = -15.9618719
        a6 = 1.80122502
        saturated_water_vapor_pressure = m.e ** (critical_temperature_kelvin / self.temperature_kelvin * (
                a1 * temp + a2 * temp ** 1.5 + a3 * temp ** 3 + a4 * temp ** 3.5 + a5 * temp ** 4 + a6 * temp ** 7.5) +
                                                 m.log(critical_pressure_pa))
        water_vapor_pressure = relative_humidity * saturated_water_vapor_pressure / 100
        mixing_ratio = 621.97 * water_vapor_pressure / (pressure_pa - water_vapor_pressure)

        # calculating atmosphere ingredients
        self.h2o = mixing_ratio / 1000
        if atmosphere == 'air':
            self.co2 = 0.000314 / (1 + self.h2o)
            self.n2 = 0.78084 / (1 + self.h2o)
            self.o2 = 0.209476 / (1 + self.h2o)
            self.ch4 = 0.000002 / (1 + self.h2o)
            self.h2 = 0.0000005 / (1 + self.h2o)
            self.o3 = 0.00000007 / (1 + self.h2o)
            self.no2 = 0.00000002 / (1 + self.h2o)
        elif atmosphere == 'nitrogen':
            self.n2 = 1 - self.h2o


[docs]    def absorption_coefficient_air(self, method, wavenumber_range, wavenumber_step, wavenumber_step_doppler):
        """
        Auxiliary method for absorption_coefficient(), designed for objects with air atmosphere, based on wavenumber notation
        - updates database basing on the newest data form hitran, checks if every necessery data is available and if not downloads it
        - with given lineshape (method) calculates proper absorption coefficient

        :param string method: method of calculating transmittance spectrum based on taken lineshape - 'voigt', 'doppler' or 'sdvoigt' (required)
        :param list wavenumber_range: wavenumber range in which transmittance spectrum should be calculated given in cm-1 (optional)
        :param float wavenumber_step: wavenumber step given in cm-1 (optional)
        :param float wavenumber_step_doppler: wavenumber step for Doppler method given in cm-1 (optional)
        """
        # database setup
        hp.db_begin('data')
        hp.db_commit()
        database_list = [['H2O', 1, 1], ['CO2', 2, 1], ['N2', 22, 1], ['O2', 7, 1], ['CH4', 6, 1], ['H2', 45, 1],
                         ['O3', 3, 1], ['NO2', 10, 1], ['CO', 5, 1], ['NH3', 7, 1]]
        for file in database_list:
            if not path.exists('data/' + file[0] + '.data') or not path.exists('data/' + file[0] + '.header'):
                hp.fetch(file[0], file[1], file[2], 200, 20000)

        # calculating absorption coefficients and transmittance spectrum for given method
        if method == 'voigt':
            nu, air_coef = hp.absorptionCoefficient_Voigt(
                SourceTables=['H2O', 'CO2', 'N2', 'O2', 'CH4', 'H2', 'O3', 'NO2'], OmegaStep=wavenumber_step,
                OmegaRange=wavenumber_range, HITRAN_units=False,
                Environment={'p': self.pressure_atm, 'T': self.temperature_kelvin},
                Diluent={'H2O': self.h2o, 'CO2': self.co2, 'N2': self.n2, 'O2': self.o2, 'CH4': self.ch4, 'H2': self.h2,
                         'O3': self.o3, 'NO2': self.no2})
        elif method == 'doppler':
            nu, air_coef = hp.absorptionCoefficient_Doppler(
                SourceTables=['H2O', 'CO2', 'N2', 'O2', 'CH4', 'H2', 'O3', 'NO2'], OmegaStep=wavenumber_step_doppler,
                OmegaRange=wavenumber_range, HITRAN_units=False,
                Environment={'p': self.pressure_atm, 'T': self.temperature_kelvin},
                Diluent={'H2O': self.h2o, 'CO2': self.co2, 'N2': self.n2, 'O2': self.o2, 'CH4': self.ch4, 'H2': self.h2,
                         'O3': self.o3, 'NO2': self.no2})
        elif method == 'sdvoigt':
            nu, air_coef = hp.absorptionCoefficient_SDVoigt(
                SourceTables=['H2O', 'CO2', 'N2', 'O2', 'CH4', 'H2', 'O3', 'NO2'], OmegaStep=wavenumber_step,
                OmegaRange=wavenumber_range, HITRAN_units=False,
                Environment={'p': self.pressure_atm, 'T': self.temperature_kelvin},
                Diluent={'H2O': self.h2o, 'CO2': self.co2, 'N2': self.n2, 'O2': self.o2, 'CH4': self.ch4, 'H2': self.h2,
                         'O3': self.o3, 'NO2': self.no2})
        else:
            raise ValueError("Method: ", method, "not defined")

        self.absorp_coef = air_coef
        self.nu = nu

        """

        - updates database basing on the newest data form hitran, checks if every necessery data is available and if not downloads it
        - with given lineshape (method) calculates proper absorption coefficient

        hp.db_begin('data')

                                                                              'T': self.temperature_kelvin},

            nu, nitrogen_coef = hp.absorptionCoefficient_SDVoigt(SourceTables=['H2O', 'N2'], OmegaStep=wavenumber_step,

                                                                 Environment={'p': self.pressure_atm,
                                                                              'T': self.temperature_kelvin},
                                                                 Diluent={'H2O': self.h2o, 'N2': self.n2})

        self.nu = nu



[docs]    def absorption_coefficient(self, method, wavelength_range_um=[2.5, 20], wavelength_step_um=0.00005,
                               wavelength_step_doppler_um=0.000005):
        """
        Method for calculating absorption coefficient, having predefined lineshape (method) and desired wavelengths
        - converts given wavelength range and wavelength step to wavenumbers

        :param string method: method of calculating transmittance spectrum based on taken lineshape - 'voigt', 'doppler' or 'sdvoigt' (required)
        :param list wavelength_range_um: wavelength range in which transmittance spectrum should be calculated given in um, default is [2.5, 20] (optional)
        :param float wavelength_step: wavelength step given in um, default is 0.00005 (optional)
        :param float wavelength_step_doppler: wavelength step for Doppler method given in um, default is 0.000005 (optional)
        """
        self.method = method
        self.wavelength_range_um = wavelength_range_um
        self.wavelength_step_um = wavelength_step_um
        self.wavelength_step_doppler_um = wavelength_step_doppler_um
        if self.method == 'doppler':
             self.wavelength_step = self.wavelength_step_doppler_um
        else:
            self.wavelength_step = self.wavelength_step_um
        wavenumber_range = [1 / self.wavelength_range_um[1] * 10000, 1 / self.wavelength_range_um[0] * 10000]
        wavenumber_range_size = wavenumber_range[1] - wavenumber_range[0]
        wavelength_range_um_size = self.wavelength_range_um[1] - self.wavelength_range_um[0]
        wavenumber_step = wavenumber_range_size * self.wavelength_step_um / wavelength_range_um_size
        wavenumber_step_doppler = wavenumber_range_size * self.wavelength_step_doppler_um / wavelength_range_um_size
        if self.atmosphere == 'air':
            self.absorption_coefficient_air(method, wavenumber_range, wavenumber_step, wavenumber_step_doppler)
        elif self.atmosphere == 'nitrogen':
            self.absorption_coefficient_nitrogen(method, wavenumber_range, wavenumber_step, wavenumber_step_doppler)



[docs]    def transmittance(self, distance_cm, desired_wavelengths):
        """
        Method returns transmittance spectrum for given environment in given wavelength range.
        Default values of wavelength step are chosen basing on hapi documentation (in wavenumber 0.01 and 0.001 respectively).
        It's not recommended to go above those values because of possible accuracy decline.
        Maximum wavelength range is [0.5, 50] um because of database which is downloaded for this range - if needed it can be changed in the code.

        :param float distance_cm: distance traveled by beam given in cm (required)
        :param numpy array desired_wavelengths: numpy array of wavelengths which the user needs, given in um (required)
        :return numpy array transmittance: values of transmittance for desired wavelengths in numpy array
        """
        wavenu, transm = hp.transmittanceSpectrum(self.nu, self.absorp_coef,
                                                  Environment={'T': self.temperature_kelvin, 'l': distance_cm})

        if self.method == 'doppler':
            wavelength, transmittance = (
                np.arange(self.wavelength_range_um[0], self.wavelength_range_um[1] + self.wavelength_step_doppler_um,
                          self.wavelength_step_doppler_um)[:transm.size], transm[::-1])
        else:
            wavelength, transmittance = (
                np.arange(self.wavelength_range_um[0], self.wavelength_range_um[1] + self.wavelength_step_um,
                          self.wavelength_step_um)[:transm.size], transm[::-1])

        return transmittance[np.searchsorted(wavelength, desired_wavelengths)]
