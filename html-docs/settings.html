Settings¶

Settings file for storing variables, some sections are influencing different files, tools are also using it. Use it with cautious.

"""
Author: Michał Porębski
"""
# plot settings
plot_type = '2d'
plot_axis_x_range = [-100, 300]
plot_axis_y_range = [-500, 100]
plot_axis_z_range = [-450, 450]
plot_how_many_pixels_shown = 10     # how many pixels are shown on 3d histogram

# characteristics paths
default_path_flat_mirror_char = "characteristics/gold_mirrors_reflectance.txt"
default_path_parabolic_mirror_char = "characteristics/gold_unprotected_unpolarized.txt"
default_path_window_char = "characteristics/window_zinc_selenide.txt"
default_path_prism_zn_se_char = "characteristics/prism-transm.txt"
default_path_screen_char = "characteristics/prism-transm.txt"

# screen settings
no_of_pixels = 128
screen_show_behind = True

# rays settings
no_of_rays = 10
show_all_rays = False
plot_rays = True

# air settings
air_temperature_celsius = 20
air_pressure_pa = 101325
air_relative_humidity = 20
air_rays_air_interaction = False

# wavelengths settings
no_of_spectrum_points = 10
wav_min = 2.5
wav_max = 20
wav_avg = (wav_max + wav_min) / 2 + 2.8

# parabolic mirror settings
no_of_parabola_surf_points = 100

# input settings
input_shape = 'gaussian'  # options: gaussian, 3gaussian, brick
input_variance = 1 * 10 ** (-6)
input_bunch_distribution_range = [-2 ** 6 * input_variance, 2 ** 6 * input_variance]
input_number_of_frames = 2 ** 14

"""Tools settings"""
tools_parabola_intersection_accuracy = 1e-10
tools_rotation_accuracy = 1e-15
